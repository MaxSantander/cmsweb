    def setUp(self):
        """
        Set up test data for the test cases.
        """
        self.client = Client()

        # Create home page
        self.home_page = Page.objects.create(
            title="Home",
            url_path="/",
            language="EN",
            meta_description="Home Description",
            meta_keywords="home, keywords",
        )
        self.home_page.publish()

        # Create test page
        self.test_page = Page.objects.create(
            title="Test Page",
            url_path="/test",
            language="EN",
            meta_description="Test Description",
            meta_keywords="test, keywords",
        )
        self.test_page.publish()

        # Create test blocks
        self.home_block = Block.objects.create(
            name="Home Block", template="<div>Home Content</div>", sorting=1
        )

        self.test_block = Block.objects.create(
            name="Test Block", template="<div>Test Content</div>", sorting=1
        )

        # Create page-block relationships
        PageBlock.objects.create(page=self.home_page, block=self.home_block)

        PageBlock.objects.create(page=self.test_page, block=self.test_block)

        # Create menu items
        MenuItem.objects.create(
            page=self.home_page, template="<li>Home Menu</li>", sorting=1
        )

        MenuItem.objects.create(
            page=self.test_page, template="<li>Test Menu</li>", sorting=2
        )

        cache.clear()

    def test_home_view(self):
        """
        Test the home page view.

        Ensure that the home page is rendered correctly with the expected content.
        """
        response = self.client.get("/")
        self.assertEqual(response.status_code, 200)
        self.assertTemplateUsed(response, "pages_app_tpl/render_page.html")
        self.assertContains(response, "Home Content")
        self.assertContains(response, "Home Menu")
