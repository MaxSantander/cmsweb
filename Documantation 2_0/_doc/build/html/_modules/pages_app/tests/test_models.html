    def test_updated_at_auto_updates(self):
        """
        Test that updated_at changes on update.

        Verifies:
            - updated_at is updated on save
            - created_at remains unchanged on save
        """
        original_updated = self.page.updated_at
        time.sleep(0.1)  # Add delay to ensure timestamp difference

        self.page.title = "Updated Title"
        self.page.save()
        self.page.refresh_from_db()

        self.assertGreater(self.page.updated_at, original_updated)
        self.assertEqual(self.page.created_at, original_updated)
