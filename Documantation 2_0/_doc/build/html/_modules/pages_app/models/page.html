"""
Page Model Module

Core content model for the CMS that handles multilingual page content.
Implements caching, SEO optimization, and version control.

Key Features:
    - Multilingual content (EN/DE)
    - SEO metadata management
    - Publication workflow
    - Cache integration
    - Historical tracking

Dependencies:
    - django.db.models
    - simple_history.models.HistoricalRecords
    - django.core.exceptions.ValidationError
    - django.utils.timezone
    - .base.TimestampedModel
"""


from django.db import models
from simple_history.models import HistoricalRecords
from .base import TimestampedModel
from django.core.exceptions import ValidationError
from django.utils import timezone

class Page(TimestampedModel):  # Inherit from TimestampedModel
    """
    Represents a dynamic page in the CMS.
    Inherits from TimestampedModel for automatic timestamp tracking.

    Fields:
        title (str): Page title
        url_path (str): URL path for the page
        name (str): Page identifier, max 100 chars
        language (str): Language code (EN/DE)
        meta_description (str): SEO meta description
        meta_keywords (str): SEO meta keywords
        published_at (datetime): When the page was published
        history (HistoricalRecords): Version tracking

    Example:
        >>> page = Page(
        ...     title="Homepage",
        ...     url_path="/home",
        ...     name="Homepage",
        ...     language="EN"
        ... )
        >>> page.save()
    """
    title = models.CharField(max_length=255)  # Add title field
    url_path = models.CharField(max_length=2048)  # Add URL path field
    name = models.CharField(max_length=100, default="Page")  # Add name field
    # Add language field German/English in 2-letter ISO code
    LANGUAGE_CHOICES = [("EN", "English"), ("DE", "Deutsch")]
    language = models.CharField(max_length=2, choices=LANGUAGE_CHOICES, default="DE")
    # SEO fields
    meta_description = models.TextField(blank=True, null=True)
    meta_keywords = models.CharField(max_length=255, blank=True, null=True)
    # Publication fields
    published_at = models.DateTimeField(blank=True, null=True)
    # Track changes to the model
    history = HistoricalRecords()

    # Add indexes for URL path and publication status
    class Meta:
        """
        Model configuration class.
        Atributes:
            indexes (list): Database indexes for optimization
        """
        indexes = [
            models.Index(fields=["url_path"]),
            models.Index(fields=["published_at"]),
        ]

    @property
    def is_published(self):
        return self.published_at is not None

    def publish(self):
        """
        Publish the page and set publication timestamp.
        """
        self.published_at = timezone.now()
        self.save()

    def unpublish(self):
        """
        Unpublish the page and clear publication timestamp.
        """
        self.published_at = None
        self.save()

    def save(self, *args, **kwargs):  # Override save method
        """
        Custom save method.
        Ensures URL path is valid before saving.
        Calls full_clean to validate the model.
        """
        self.full_clean()
        super().save(*args, **kwargs)

    def clean(self):
        """
        Custom model validation.
        Ensures URL path starts with a forward slash.
        Raises:
            ValidationError: If URL path is invalid
        """
        if not self.url_path.startswith("/"):
            raise ValidationError(
                {"url_path": "URL path must start with a forward slash (/)"}
            )



    def __str__(self):
        """
        String representation of the page.

        Returns:
            str: Format: "{title} - {language}"
        """
        return f"{self.title} - {self.language}"
