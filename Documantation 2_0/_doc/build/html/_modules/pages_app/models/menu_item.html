"""
MenuItem Model Module

This module defines the MenuItem model which represents navigation items in the CMS.
MenuItems are linked to pages and can be sorted.

Key Features:
    - Linked to pages
    - Template-based content
    - Sorting order
    - Historical tracking

Dependencies:
    - django.db.models
    - simple_history.models.HistoricalRecords
    - .base.TimestampedModel
    - .page.Page
"""

from django.db import models
from .base import TimestampedModel
from .page import Page
from simple_history.models import HistoricalRecords



[docs]
class MenuItem(TimestampedModel):
    """
    Represents a navigation item in the CMS.

    Attributes:
        page (ForeignKey): Reference to the linked page
        name (str): Name of the menu item
        template (str): Template content for the menu item
        sorting (int): Sorting order
        history (HistoricalRecords): Tracks changes to the menu item

    Examples:
        >>> page = Page.objects.get(slug="home")
        >>> menu_item = MenuItem.objects.create(
        ...     page=page,
        ...     name="Home",
        ...     template="<a href='/home'>Home</a>",
        ...     sorting=1
        ... )
    """
    page = models.ForeignKey(Page, on_delete=models.CASCADE)
    name = models.CharField(max_length=100, default="Menu Item")
    template = models.TextField()
    sorting = models.IntegerField()
    history = HistoricalRecords()  # track changes to the model

    class Meta:
        indexes = [
            models.Index(fields=["sorting"]),
        ]

    def __str__(self):
        return f"{self.name} ({self.page.language})"
